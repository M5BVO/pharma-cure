\begin{presentercard}{CARD 2: RELATIONAL CORE \& INTEGRITY}
    \begin{minipage}{0.62\textwidth}
        \textbf{[CONCEPT]} "Highly normalized data structure with strict referential integrity."
        \begin{itemize}
            \item \textbf{3rd Normal Form (3NF):} Eliminates data redundancy and ensures logical consistency across actors.
            \item \textbf{Enforced Logic:} Uses \texttt{ON DELETE CASCADE} to ensure medical records are purged when user accounts are closed.
            \item \textbf{Role Segregation:} Strict \texttt{ENUM} constraints enforce access control for Admins, Pharmacists, and Patients.
        \end{itemize}
    \end{minipage}

\begin{presentercard}{CARD 3: ATOMIC MARKET TRANSACTIONS}
    \textbf{[CONCEPT]} "Ensuring the 'ACID' properties of the clinical marketplace."
    \begin{itemize}
        \item \textbf{Transactional Atomicity:} Uses \texttt{beginTransaction()} so that restocks and payment logs succeed or fail as a single atomic unit.
        \item \textbf{Uber-Handshake:} Replaces static lists with a dynamic Bidding State Machine (Pending $\to$ Accepted/Declined).
        \item \textbf{Proximity Awareness:} System ranks providers based on distance to optimize clinical delivery time.
    \end{itemize}
\end{presentercard}

\vspace{0.4cm}

\begin{presentercard}{CARD 4: THE SECURITY FORTRESS}
    \begin{minipage}{0.62\textwidth}
        \textbf{[CONCEPT]} "Stateless, cryptographic defense of patient data."
        \begin{itemize}
            \item \textbf{Bcrypt Hashing:} Passwords are never stored. We store a 60-character salted hash with 10 work rounds to block brute-force attacks.
            \item \textbf{JWT Authentication:} Uses a Header-Payload-Signature structure. This enables stateless verification without needing server-side session memory.
        \end{itemize}
    \end{minipage}
    \begin{minipage}{0.35\textwidth}
        \centering
        \begin{tikzpicture}[scale=0.7, every node/.style={transform shape}]

\end{presentercard}
